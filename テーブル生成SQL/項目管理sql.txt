CREATE TABLE `project` (
  `id` int NOT NULL AUTO_INCREMENT COMMENT '主キー',
  `pro_no` varchar(10) NOT NULL COMMENT '項目番号',
  `pro_name` varchar(20) NOT NULL COMMENT '項目名',
  `pro_address` varchar(100) NOT NULL COMMENT '項目住所',
  `cust_name` varchar(20) NOT NULL COMMENT '顧客名',
  `emp_name` varchar(20) NOT NULL COMMENT '社員名',
  `en_time` date NOT NULL COMMENT '入場時間',
  `exit_time` date NOT NULL COMMENT '退場時間',
  PRIMARY KEY (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=5 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci
